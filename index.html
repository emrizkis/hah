def y(a, xnew, reg):
   
    # get coefisien for linear regresion
    coef1 = np.polyfit(X,Y1,reg)
    coef2 = np.polyfit(X,Y2,reg)
    coef3 = np.polyfit(X,Y3,reg)

    LEIG = np.poly1d(coef1) 
    LN   = np.poly1d(coef2)
    WB   = np.poly1d(coef3)
    
    if (a==1):
        return LEIG(xnew)
    elif (a==2):
        return LN(xnew)
    else:
        return WB(xnew)
